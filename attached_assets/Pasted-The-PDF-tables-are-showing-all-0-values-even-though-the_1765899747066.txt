The PDF tables are showing all $0 values even though the app displays correct numbers.

The issue is that the markdown tables in full_report_markdown aren't rendering correctly when converted to PDF.

Instead of relying on the LLM-generated markdown, we should build the tables ourselves from the JSON data when creating the PDF.

Update pdf_generator.py to:

1. Accept the full report_data JSON (not just markdown string)
2. Build clean HTML tables from the JSON data
3. Use the executive_summary and narratives from JSON
4. Generate proper HTML tables with the actual numbers

New function signature:
```python
def generate_pdf_from_markdown(report_data: dict, project_name: str) -> bytes:
    """
    Generate PDF from report JSON data
    
    Args:
        report_data: Full report JSON with all sections
        project_name: Name of the project
    """
    
    # Build HTML sections from JSON
    html_sections = []
    
    # Executive Summary
    html_sections.append(f"<h2>Executive Summary</h2>")
    html_sections.append(f"<p>{report_data.get('executive_summary', '')}</p>")
    
    # Key Metrics
    html_sections.append(f"<h2>Key Metrics</h2>")
    html_sections.append(f"<p><strong>Year 1 CRA Revenue:</strong> ${report_data['fiscal_highlights']['year_1_cra_revenue']:,.0f}</p>")
    html_sections.append(f"<p><strong>10-Year Cumulative:</strong> ${report_data['fiscal_highlights']['ten_year_cumulative']:,.0f}</p>")
    
    # CRA Increment Table
    html_sections.append(f"<h2>Fiscal Impact: CRA Tax Increment</h2>")
    html_sections.append("<table>")
    html_sections.append("<tr><th>Year</th><th>Taxable Value</th><th>CRA Increment</th><th>Cumulative</th></tr>")
    for row in report_data.get('cra_increment_projection', []):
        html_sections.append(f"<tr>")
        html_sections.append(f"<td>{row['year']}</td>")
        html_sections.append(f"<td>${row['taxable_value']:,.0f}</td>")
        html_sections.append(f"<td>${row['cra_increment']:,.0f}</td>")
        html_sections.append(f"<td>${row['cumulative']:,.0f}</td>")
        html_sections.append(f"</tr>")
    html_sections.append("</table>")
    
    # Construction Impact
    construction = report_data.get('construction_impact', {})
    html_sections.append(f"<h2>Construction Phase (One-Time)</h2>")
    html_sections.append(f"<p>{construction.get('narrative', '')}</p>")
    html_sections.append("<table>")
    html_sections.append("<tr><th>Impact Type</th><th>Output</th><th>Jobs</th><th>Labor Income</th></tr>")
    for row in construction.get('table', []):
        html_sections.append(f"<tr>")
        html_sections.append(f"<td>{row['impact_type']}</td>")
        html_sections.append(f"<td>${row['economic_output']:,.0f}</td>")
        html_sections.append(f"<td>{row['jobs']:.1f}</td>")
        html_sections.append(f"<td>${row['earnings']:,.0f}</td>")
        html_sections.append(f"</tr>")
    html_sections.append("</table>")
    
    # Operations Impact
    operations = report_data.get('operations_impact', {})
    html_sections.append(f"<h2>Operations Phase (Recurring Annual)</h2>")
    html_sections.append(f"<p>{operations.get('narrative', '')}</p>")
    html_sections.append("<table>")
    html_sections.append("<tr><th>Impact Type</th><th>Output</th><th>Jobs</th><th>Labor Income</th></tr>")
    for row in operations.get('table', []):
        html_sections.append(f"<tr>")
        html_sections.append(f"<td>{row['impact_type']}</td>")
        html_sections.append(f"<td>${row['economic_output']:,.0f}</td>")
        html_sections.append(f"<td>{row['jobs']:.1f}</td>")
        html_sections.append(f"<td>${row['earnings']:,.0f}</td>")
        html_sections.append(f"</tr>")
    html_sections.append("</table>")
    
    # Community Impacts
    html_sections.append(f"<h2>Community Impacts</h2>")
    for impact in report_data.get('community_impacts', []):
        html_sections.append(f"<h3>{impact['category']}</h3>")
        html_sections.append(f"<p>{impact['description']}</p>")
    
    # Join all sections
    content_html = "
".join(html_sections)
    
    # Wrap in template with CSS (keep existing CSS)
    html_template = f"""
    <!DOCTYPE html>
    <html>
    <head>
        <meta charset="UTF-8">
        <title>Economic Impact Report - {project_name}</title>
        <style>
            /* Keep your existing CSS */
        </style>
    </head>
    <body>
        <h1>Economic Impact Report - {project_name}</h1>
        {content_html}
        <div class="footer">
            <p>Powered by Street Economics | BusinessFlareÂ® Analytics</p>
        </div>
    </body>
    </html>
    """
    
    html = HTML(string=html_template)
    return html.write_pdf()
```
